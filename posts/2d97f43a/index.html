유의

통계는 과거의 산물이며 역사는 반복되지 않을 수 있다. 표본이 충분하지 않은 사건의 통계에 과도한 의미 부여를 해서는 안된다.

# 미국 국채 10년물 금리 인하 기간에서 주요 인덱스 지수 수익율 비교

# 기간 내 금리 하락 구간 목록 찾기
def get_gs10_fall_periods(start_date, end_date):

    # FRED 에서 금리 데이터 가져오기 (10년 만기 미국 국채 금리)
    gs10 = pdr.get_data_fred("GS10", start=start_date, end=end_date)

    periods = [
        # ("2023-01-01", "2023-07-01"),
        # ("2023-07-01", "2023-12-14"),
    ]
    previous_date = None
    previous_rate = None
    fall_start_date = None
    fall_start_rate = None
    for index, row in gs10.iterrows():
        date = index  # .to_pydatetime().date()
        rate = row["GS10"]
        if previous_rate is not None:
            # 하락 추이
            if rate < previous_rate:
                # 하락 초입이면 전 날짜를 시작으로 선정
                if fall_start_date is None:
                    fall_start_date = previous_date
                    fall_start_rate = previous_rate
            # 상승 전환
            elif rate > previous_rate:
                # 하락 초입 시작 날짜가 있으면 그때부터 저번 구간까지가 하락 구간이므로 추가
                if fall_start_date is not None:
                    # print(f"금리 하락 구간: {fall_start_date} ~ {previous_date} [{fall_start_rate:.2f} -> {previous_rate:.2f}]")
                    periods.append((fall_start_date, previous_date))
                    fall_start_date = None
                    fall_start_rate = None

        previous_date = date
        previous_rate = rate

    intervals = pd.IntervalIndex.from_tuples(periods, closed="left")
    return intervals


# 주가 데이터 측정 기간
history_start = pd.Timestamp("1989-12-01")
history_end = pd.Timestamp("2023-12-15")

def get_history(ticker, start_date, end_date):

    key = f"{ticker.ticker}@{str(start_date)}~{str(end_date)}"
    result = cache.get(key)

    if result is None:
        print(f"{key} is not cached. so i'll 'try to load from web.")
        result = ticker.history(period="1d", start=start_date, end=end_date)
        cache[key] = result

    result.index = result.index.tz_localize(None)
    return result


def calc_returns(ticker, intervals, trading_days_in_year=252):

    # print(f"calc_returns: {ticker.ticker}, {intervals}")

    alls = get_history(ticker, history_start, history_end)

    alls["Daily Changes"] = alls["Close"].pct_change()

    trading_days_in_year = 252 if trading_days_in_year is None else trading_days_in_year

    filtered = alls[
        alls.index.to_series().apply(lambda idx: intervals.contains(idx).any())
    ]

    if filtered.empty:
        print(f"{ticker.ticker:6} 해당 기간내 필터링된 데이터가 없습니다. ex) 금리 인하 기간이 없습니다.")
        return

    daily_changes = filtered["Daily Changes"]
    
    up_count = (daily_changes > 0).sum()

    daily_changes_plus_1 = daily_changes.dropna().apply(lambda x: x + 1)
    period_changes_prod = daily_changes_plus_1.prod() - 1
    daily_geomean = (period_changes_prod + 1) ** (1 / len(daily_changes_plus_1)) - 1
    monthlized_returns = ((daily_geomean + 1) ** 21) - 1
    annualized_returns = ((daily_geomean + 1) ** trading_days_in_year) - 1

    print(
        f"{ticker.ticker:6} 필터기간 수익율: "
        f"P {period_changes_prod * 100:+9.3f}% / "
        f"D {daily_geomean * 100:+7.3f}% / "
        f"M {monthlized_returns * 100:+7.3f}% / "
        f"Y {annualized_returns * 100:+9.3f}% | "
        f"거래일수: {len(daily_changes):4} -> {len(daily_changes_plus_1):4} | 상승일수: {up_count:4} | 상승비율: {(up_count/len(daily_changes))*100:.2f}%"
    )


print("[[[ 금리 인하 기간 (및 주위 기간) 주요 인덱스 수익율 비교 ]]]")
print(f"전체 기간: {history_start} ~ {history_end}")

def offset_interval(interval, offset, period=None):
    left = interval.left + offset
    right = interval.right + offset if period is None else left + period
    return pd.Interval(left, right)


def show_graph(tickers):
    plt.figure(figsize=(12, 6))
    plt.yscale("log", base=2)
    for ticker in tickers:
        # name = ticker.ticker
        name = ticker.info["shortName"]

        plt.plot(history.index, history["Close"], label=name, linewidth=2)
    plt.title(f"{' vs '.join(map(lambda x: x.ticker, tickers))} Closing Prices")
    plt.xlabel("Date")
    plt.ylabel("Price (USD)")
    plt.legend()
    plt.grid()
    plt.show()

for ticker in tickers:
    print(f"{ticker.ticker:<6} {ticker.info['shortName']}")
### 전체 기간

full_periods = pd.IntervalIndex.from_tuples(
    [(history_start, history_end)], closed="left"
)

for ticker in tickers:
    calc_returns(ticker, full_periods, None)

### 금리 인하 기간 수익율 출력 (기간 오프셋별로 모두 출력)
def remains():
    print("#### 주가 수익율 for 10년물 국채 금리 하락기 ####")
    gs10_fall_periods = get_gs10_fall_periods(fall_check_start, fall_check_end)

    for offset_weeks in range(-52, +56, 4):
        period_weeks = 4

        offset = pd.Timedelta(weeks=offset_weeks)
        period = pd.Timedelta(weeks=period_weeks)
        # period = None # if 원본 길이에 맞춤

        print(f">> 오프셋 {offset_weeks}wk | 기간 {period_weeks}wk")

        offset_periods = gs10_fall_periods.map(
            lambda interval: offset_interval(interval, offset=offset, period=period)
        )

        for ticker in tickers:
            calc_returns(ticker, offset_periods, None)


t = threading.Thread(target=remains, args=())
t.start()

# show_graph(tickers)

t.join()

미국채 10년물 금리는 FRED 를 통해 구하였고 pandas 를 통해 조회 가능하다.

아래는 필터기간 수익율에 대한 설명이다. 오프셋 - 해당 필터기간은 모든 금리 하락 기간 시작점에서 얼마나 더했는지를 말해준다. 기간 - 해당 필터기간의 길이를 말해준다.

즉 필터기간의 범위는

시작: 금리하락 시점 + 오프셋

끝: 금리하락 시점 + 오프셋 + 기간까지 조회된다.

P - 해당 필터기간 총 수익율을 말한다.

D - 해당 필터기간 일일 평균 수익율을 말한다.

M - D 를 월간 수익율로 환산한 값이다. (월 평균 거래일수 21일 이용)

Y - D 를 연간 수익율로 환산한 값이다. (연 평균 거래일수 252일 이용)

거래일수 - 해당 필터기간의 총 일수를 말한다. 오프셋 차이만 있으므로 차이가 없어야 하지만 휴일이 포함되거나 오프셋으로 인해 전체 조회 기간을 벗어날 경우 데이터가 없기 때문에 그만큼 거래일 수에서 차이를 보일 수 있다. 상승일수 - 필터기간 거래일수 중에서 지수가 상승한 일수를 말한다. 상승비율 - 상승일수 / 거래일수

끝내며

이를 토대로 금리 인하 기간 전후로 나스닥, 가치주, 소형주에 유의미한 수익율 추이 변화를 느끼지는 못했다. 일부 확연히 차이가 나는 기간이 있지만 통계 표본 부족에 의한 오차인지 시장의 끌림 현상인지 구분하기 어렵다. 무엇보다도 언제 그러한 차이가 발생되는지 예측하는 것은 불가능에 가깝다. 주식 시장의 근본적인 원리일지 단지 기계적인 리밸런싱 과정상의 이슈일지 판단하기란 더 어렵다. 물론 필요한 경우 다른 인덱스 티커를 사용하여 다른 추정을 끌어낼 수도 있다.

하지만 앞서 언급했지만 큰 의미 부여를 해서는 안된다. 과거는 반복되지 않으며 과거는 단지 확률에서 현실에 벌어진 기록만 남겨졌다는 것을 기억하자. 최소한 오늘날까지는 주식 분석 기술에서 멀티유니버스를 시뮬레이션할 방법은 없다. 따라서 과거 분석을 통해 무작정 인과관계를 도출해서는 안된다. 퀀트와 마찬가지다. 단지 금리 인하 기간을 전후로 어떠한 변동성을 가졌는지 정도가 근소하게 유의한 정보일 수 있다.

만일 인과관계 분석을 원하는 사람이 있다면 해당 기간 전후로 어떠한 뉴스가 있었는지도 검토해야 한다.
